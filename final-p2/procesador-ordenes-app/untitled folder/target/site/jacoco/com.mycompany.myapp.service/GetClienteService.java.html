package com.mycompany.myapp.service;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.web.client.RestTemplate;

@Service
@Transactional
public class GetClienteService {

    private final Logger log = LoggerFactory.getLogger(GetClienteService.class);

    private final RestTemplate restTemplate;


    @Value("http://192.168.194.254:8000/api/ordenes/ordenes")
    private String url;

    public GetClienteService(RestTemplate restTemplate) {
        this.restTemplate = restTemplate;
    }

    public String getOrders() {
        return restTemplate.getForObject(url, String.class);
    }



}
